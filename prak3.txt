Public Class Ordermakanan
    Public hrgmakanan, hrgminuman As Integer
    Private Sub cbomkn_SelectedIndexChanged(ByVal sender As System.Object, ByVal e As System.EventArgs) Handles cbomkn.SelectedIndexChanged
        Select Case cbomkn.Text
            Case "Nasi Uduk"
                Label6.Text = "3500"
                hrgmakanan = 3500
            Case "Bakso"
                Label6.Text = "6000"
                hrgmakanan = 6000
            Case "Mie Ayam"
                Label6.Text = "7000"
                hrgmakanan = 7000
        End Select
    End Sub

    Private Sub cbomnm_SelectedIndexChanged(ByVal sender As System.Object, ByVal e As System.EventArgs) Handles cbomnm.SelectedIndexChanged
        Select Case cbomnm.Text
            Case "Es Jeruk"
                Label7.Text = "3000"
                hrgminuman = 3000
            Case "Jus Alpukat"
                Label7.Text = "7000"
                hrgminuman = 7000
            Case "Teh Manis"
                Label7.Text = "4000"
                hrgminuman = 4000
        End Select
    End Sub
    Public Sub New()

        ' This call is required by the designer.
        InitializeComponent()
        cbomnm.Visible = False
        Label3.Visible = False

        ' Add any initialization after the InitializeComponent() call.

    End Sub

    Private Sub CheckBox1_CheckedChanged(ByVal sender As System.Object, ByVal e As System.EventArgs) Handles CheckBox1.CheckedChanged
        If CheckBox1.Enabled = False Then
            cbomnm.Visible = False
        End If

        If CheckBox1.Enabled = True Then
            cbomnm.Visible = True
        End If
    End Sub

        porsimnm = txtporsimnm.Text
